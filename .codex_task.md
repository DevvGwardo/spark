## Task 2: Add Skills Hub Browse/Install to the sidebar

Add a "Browse Hub" feature to the existing HermesSkillsPanel so users can discover and install skills from the Hermes Skills Hub.

### Changes needed:

#### 2a. Add API functions to `src/lib/hermes-api.ts`

Add these interfaces and functions:

```typescript
export interface HubSkill {
  name: string;
  description: string;
  category: string;
  source: 'built-in' | 'optional' | 'community' | 'anthropic' | 'lobehub';
  installed: boolean;
}

export async function fetchSkillsHub(): Promise<HubSkill[]> {
  const data = await hermesFetch<{ skills: HubSkill[] }>('/workspace/skills/hub');
  return data.skills ?? [];
}

export async function installHubSkill(skillName: string): Promise<void> {
  await hermesFetch('/workspace/skills/hub/install', {
    method: 'POST',
    body: JSON.stringify({ name: skillName }),
  });
}
```
